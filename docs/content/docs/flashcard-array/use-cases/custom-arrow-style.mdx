## Code Example

```tsx
'use client'

import { FlashcardArray, IFlashcard, useFlashcardArray } from 'react-quizlet-flashcard'
import 'react-quizlet-flashcard/dist/index.css'

export default function CyclingArray() {
  const deck: IFlashcard[] = [...]

  const flipArrayHook = useFlashcardArray({
    deckLength: deck.length,
  })

  return (
    <FlashcardArray
      flipArrayHook={flipArrayHook}
      deck={deck}
      style={
        {
          "--prev-arrow-color": "#2ec140",
          "--next-arrow-color": "#2ec140",
          "--disabled-arrow-color": "#c3a3e6",
        } as CSSProperties
      }
    />
  )
}
```
